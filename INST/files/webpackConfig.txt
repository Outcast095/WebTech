import path from 'path';
import { fileURLToPath } from 'url';
import HtmlWebpackPlugin from 'html-webpack-plugin';
import MiniCssExtractPlugin from 'mini-css-extract-plugin';
import TerserPlugin from 'terser-webpack-plugin';
import CssMinimizerPlugin from 'css-minimizer-webpack-plugin';

// Эмуляция переменных Node.js (так как мы используем ES Modules тип в JS)
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// Флаг режима: true если разработка, false если продакшен
const isDev = process.env.NODE_ENV === 'development';

export default {
  // mode: Включает встроенные оптимизации Webpack (сжатие для prod, скорость для dev)
  mode: isDev ? 'development' : 'production',

  // entry: Точка входа, откуда Webpack начинает строить дерево зависимостей
  entry: './src/index.js',

  // output: Инструкции, куда и как сохранить итоговые файлы
  output: {
    path: path.resolve(__dirname, 'dist'),
    // [contenthash]: Добавляет уникальный хеш контента в имя файла для сброса кэша браузера
    filename: isDev ? 'js/[name].js' : 'js/[name].[contenthash].js',
    clean: true, // Очищает папку dist перед каждым билдом
  },

  // devtool: Настройка Source Maps (позволяет видеть оригинальный код в консоли браузера при ошибках)
  devtool: isDev ? 'eval-source-map' : 'source-map',

  // devServer: Настройки локального сервера для разработки
  devServer: {
    static: { directory: path.resolve(__dirname, 'dist') },
    compress: true,   // Gzip сжатие передаваемых данных
    port: 3000,       // Порт сервера
    open: true,       // Авто-открытие браузера
    hot: true,        // Hot Module Replacement (обновление без перезагрузки страницы)
    historyApiFallback: true, // Нужно для Single Page Applications (React Router), чтобы не было 404
  },

  module: {
    rules: [
      // 1. JS/JSX: Прогоняем через Babel (транспиляция нового кода в старый + JSX)
      {
        test: /\.(js|jsx)$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            presets: [
              '@babel/preset-env',
              ['@babel/preset-react', { runtime: 'automatic' }]
            ]
          }
        }
      },
      // 2. Стили: SASS -> PostCSS (автопрефиксы) -> CSS -> В JS или отдельный файл
      {
        test: /\.(s[ac]ss|css)$/i,
        use: [
          // В dev-режиме вставляем стили в <style>, в prod — выносим в .css файл для скорости
          isDev ? 'style-loader' : MiniCssExtractPlugin.loader,
          'css-loader',
          {
            loader: 'postcss-loader',
            options: {
              postcssOptions: { plugins: [['autoprefixer', {}]] },
            },
          },
          'sass-loader', // Компилирует SCSS/SASS в CSS
        ],
      },
      // 3. Assets: Картинки (копируются в dist/assets/images с хешем)
      {
        test: /\.(png|svg|jpg|jpeg|gif)$/i,
        type: 'asset/resource',
        generator: { filename: 'assets/images/[name][hash][ext]' },
      },
      // 4. Assets: Шрифты
      {
        test: /\.(woff|woff2|eot|ttf|otf)$/i,
        type: 'asset/resource',
        generator: { filename: 'assets/fonts/[name][ext]' },
      }, 
    ],
  },

  plugins: [
    // Генерирует index.html и сам подключает туда бандлы
    new HtmlWebpackPlugin({ template: './src/index.html' }),
    // Вырезает CSS из JS-бандла в отдельные файлы (улучшает загрузку)
    new MiniCssExtractPlugin({ filename: 'css/[name].[contenthash].css' }),
  ],

  resolve: {
    // Позволяет не писать .js или .jsx при импортах
    extensions: ['.js', '.jsx']
  },

  // optimization: Тонкая настройка производительности
  optimization: {
    minimize: !isDev, // Минификация включается только для билда
    minimizer: [
      new TerserPlugin(),       // Сжимает JavaScript
      new CssMinimizerPlugin(), // Сжимает CSS (удаляет пробелы, дубликаты)
    ],
    // splitChunks: Выносит общие библиотеки (React, и т.д.) в отдельный файл vendor.js
    splitChunks: { chunks: 'all' },
  },
};
