1.// npm init -y / инициализация проекта 

в файле package.json заменяем "type": "commonjs" на "type": "module"


2.//  Установка и настройка: Установка webpack и webpack-cli. Создание файла webpack.config.js.
npm install webpack webpack-cli --save-dev



3.// установка cross-env  Это гарантирует, что проект будет одинаково успешно собираться на Windows, и на любой другой системе.
npm install --save-dev cross-env

настройка файла package.json

"scripts": {
  "start": "cross-env NODE_ENV=development webpack serve",
  "build": "cross-env NODE_ENV=production webpack --mode production"
},

3. // настройка файла webpack.config.js
import path from 'path'; // Стандартный модуль Node.js для работы с путями. Нужен, чтобы пути корректно работали и на Windows, и на Linux.


import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const isDev = process.env.NODE_ENV !== 'production';

export default {
  mode: isDev ? 'development' : 'production',

  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'bundle.js',
    clean: true,
  },

  plugins: [],

  module: {
    rules: []
  },

  devServer: {},
  optimization: {},
  resolve: {},


};
